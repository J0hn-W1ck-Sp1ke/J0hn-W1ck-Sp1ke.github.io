KMP算法

前缀函数和KMP

#include <iostream>
#include <cstring>
using namespace std;
const int N=1e6+10;
int nxt[N];
char text[N], pattern[N];

void getNext(const char* s)//求模式串的前缀函数
{
    int k=0;
    int n=strlen(s);
    for(int i=1; i<n; ++i)
    {
        while(k>0 && s[i]!=s[k]) k=nxt[k-1];
        if(s[i]==s[k]) ++k;
        nxt[i]=k;
        //cout<<nxt[i]<<" ";
    }
    //cout<<"\n";
}

void KMP()//匹配模式串和文本串
{
    getNext(pattern);
    int k=0;//k表示匹配的长度
    int tlen=strlen(text);
    int plen=strlen(pattern);
    for(int i=0; i<tlen; ++i)
    {
        //text和pattern不匹配，且k>0，表明text和patt有部分匹配
        while(k>0 && pattern[k] != text[i]) k=nxt[k-1];//往前回溯
        if(pattern[k] == text[i]) ++k;
        //cout<<k<<" ";
        if(k == plen)//k移动到pattern末端
        {
            cout<<i-plen+1+1<<endl;//输出位置
            k=nxt[k-1];//重新初始化，寻找下一个
            //i=i-plen+1;//i定位到该位置，继续寻找下一个
            //return i-plen+1;
        }
    }
    //cout<<"\n";
    //return -1;
}

int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    cin>>text;
    cin>>pattern;
    KMP();
    for(int i=0; i<strlen(pattern); ++i)
        cout<<nxt[i]<<" ";
    cout<<"\n";
}
